# Dynamic Programming

#### Number of Paths
Test a driverless car navigating from the bottom-left to the top-right of an nÃ—n grid. Write a function numOfPathsToDest(n) that returns the number of possible paths the car can take.
```js
/*
Keywords: Dynamic Programming, Grid traversal, Bottom-up approach, 2D array, Nested loop, Memoization table, Row, Column, Subproblems, Path counting
*/

function numOfPathsToDest(n) {
  // your code goes here
  let moves = Array.from({length: n}, el => new Uint32Array(n))
  //let moves = [[1]]

  moves[0][0]=1

  for(let i = 1; i < n; i++){

    for(let j = 0; j <= i; j++){

      if(i -1 >= j) {
        moves[i][j] = moves[i - 1][j]
      }

      if(j - 1 >= 0) {
        moves[i][j] = moves[i][j] + moves[i][j - 1]
      }
    }
  }

  return moves[n -1][n -1]
}
```
